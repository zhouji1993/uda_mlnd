""" Gaussian Jordan  Ax = b.
    
        A:  
        b: 
        decPts: 4
        epsilon: 0 1.0e-16
        
     x  Ax = b 
    None Ab 
    None A 
"""
def gj_Solve(A, b, decPts=4, epsilon=1.0e-16):
    if len(A) != len(b) : return None # 
    Ab = augmentMatrix(A, b)

    for g in range(len(A)):
        # 
        col = transpose(Ab)[g][g:] 
        max_val = max(col, key=abs)
        max_idx = col.index(max_val) + g 
        if abs(max_val) < epsilon: return None # 

        # 
        swapRows(Ab, g, max_idx) 
        scaleRow(Ab, g, 1.0/Ab[g][g])

        # 
        for i in range(len(Ab)):
            if i != g and Ab[i][g] != 0:
                addScaledRow(Ab,i,g,-Ab[i][g])

    result = [[n[-1]] for n in Ab]
    matxRound(result, decPts)

    return result
